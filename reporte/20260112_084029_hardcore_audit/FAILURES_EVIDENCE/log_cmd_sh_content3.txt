#!/usr/bin/env bash
set -euo pipefail
LOG_FILE="${LOG_FILE:-./reporte/20260112_084029_hardcore_audit/FAILURES_EVIDENCE/command_log.csv}"
run_and_log() {
  local out_file="$1"
  shift
  local cmd=("$@")
  local start_ts end_ts duration exit_code
  start_ts=$(date -u +%Y-%m-%dT%H:%M:%SZ)
  local start_ms
  start_ms=$(date +%s%3N)
  { "${cmd[@]}"; } 2>&1 | tee "$out_file"
  exit_code=${PIPESTATUS[0]}
  local end_ms
  end_ms=$(date +%s%3N)
  duration=$((end_ms - start_ms))
  printf "%s,%s,%s,"%s",%s
" "$start_ts" "$duration" "$exit_code" "${cmd[*]}" "$out_file" >> "$LOG_FILE"
  return $exit_code
}
